626  bool RayTrace(int x1, int y1, int x2, int y2) {
627  int dx = Mathf.Abs(x2 - x1);
628  int sx = -1;
629 
630  if(x1 < x2) {
631  sx = 1;
632  }
633 
634  int dy = Mathf.Abs(y2 - y1);
635  int sy = -1;
636 
637  if(y1 < y2) {
638  sy = 1;
639  }
640 
641  int err = -dy/2;
642 
643  if(dx > dy) {
644  err = dx/2;
645  }
646 
647  while(true) {
648 
649  if(!_tiles[x1, y1].baseTerrain.walkable) {
650  return false;
651  }
652 
653  if(x1 == x2 && y1 == y2) {
654  return true;
655  }
656 
657  int e2 = err;
658 
659  if(e2 > -dx) {
660  err -= dy;
661  x1 += sx;
662  }
663 
664  if(e2 < dy) {
665  err += dx;
666  y1 += sy;
667  }
668  }

749  break;

756  }
