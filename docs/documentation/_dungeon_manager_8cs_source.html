639  void SortTerrainObjects() {
640 
641  List<TerrainBehaviour> terrainObjects = new List<TerrainBehaviour>();
642 
643  // Get a list of valid terrain objects
644  for(int i = 0; i < _terrainLayer.transform.childCount; i++) {
645  TerrainBehaviour tb = _terrainLayer.transform.GetChild(i).GetComponent<TerrainBehaviour>();
646  if(tb != null) {
647  if(tb.tile != null) {
648  terrainObjects.Add(tb);
649  } else {
650  tb.transform.SetAsFirstSibling();
651  }
652  }
653  }
654 
655  // Sort terrain by depth (ascending insertion sort)
656  if(terrainObjects.Count >= 2) {
657  int i = 1;
658  while(i < terrainObjects.Count) {
659  int j = i;
660 
661  while(j > 0 && (dimension * terrainObjects[j - 1].tile.position.y + terrainObjects[j - 1].tile.position.x) > (dimension * terrainObjects[j].tile.position.y + terrainObjects[j].tile.position.x)) {
662  int siblingNdx = terrainObjects[j - 1].transform.GetSiblingIndex();
663  TerrainBehaviour tempBehaviour = terrainObjects[j - 1];
664 
665  // Swap transform depth
666  terrainObjects[j - 1].transform.SetSiblingIndex(terrainObjects[j].transform.GetSiblingIndex());
667  terrainObjects[j].transform.SetSiblingIndex(siblingNdx);
668 
669  // Swap behaviour order
670  terrainObjects[j - 1] = terrainObjects[j];
671  terrainObjects[j] = tempBehaviour;
672 
673  j--;
674  }
675  i++;
676  }
677  }
