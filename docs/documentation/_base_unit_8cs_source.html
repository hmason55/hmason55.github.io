955  public List<PathNode> FindPath(Vector2Int startPosition, Vector2Int endPosition, bool ignoreUnits = false, int excludeNodesFromEnd = 0) {
956  PathNode currentNode = null;
957  PathNode startNode = new PathNode(startPosition);
958  PathNode endNode = new PathNode(endPosition);
959  List<PathNode> openNodes = new List<PathNode>();
960  List<PathNode> closedNodes = new List<PathNode>();
961  bool[,] walkableTiles = GetWalkableTiles(startPosition, endPosition, ignoreUnits);
962  int distanceFromStart = 0;
963  openNodes.Add(startNode);
964 
965  while(openNodes.Count > 0) {
966  int lowest = int.MaxValue;
967  foreach(PathNode node in openNodes) {
968  int distance = node.totalDistance;
969  if(distance < lowest) {
970  lowest = distance;
971  currentNode = node;
972  }
973  }
974 
975  closedNodes.Add(currentNode);
976  openNodes.Remove(currentNode);
977 
978  foreach(PathNode node in closedNodes) {
979  if( node.position.x == endNode.position.x &&
980  node.position.y == endNode.position.y) {
981  goto createPath;
982  }
983  }
984 
985  List<PathNode> neighborNodes = GetWalkableNeighborNodes(currentNode.position, endPosition, walkableTiles,ignoreUnits);
986  distanceFromStart++;
987 
988  foreach(PathNode neighborNode in neighborNodes) {
989 
990  // If closed nodes already contain this node then skip it
991  foreach(PathNode node in closedNodes) {
992  if( node.position.x == neighborNode.position.x &&
993  node.position.y == neighborNode.position.y) {
994  goto nextNeighbor;
995  }
996  }
997 
998  // Check if it's in the open list
999  bool inOpenNodes = false;
1000  foreach(PathNode node in openNodes) {
1001  if( node.position.x == neighborNode.position.x &&
1002  node.position.y == neighborNode.position.y) {
1003  inOpenNodes = true;
1004  break;
1005  }
1006  }
1007 
1008  if(inOpenNodes) {
1009  // Check if this is a better path
1010  if(distanceFromStart + neighborNode.trueDistanceFromEnd < neighborNode.totalDistance) {
1011  neighborNode.distanceFromStart = distanceFromStart;
1012  neighborNode.totalDistance = neighborNode.distanceFromStart + neighborNode.trueDistanceFromEnd;
1013  neighborNode.parentNode = currentNode;
1014  }
1015  } else {
1016  neighborNode.distanceFromStart = distanceFromStart;
1017  neighborNode.trueDistanceFromEnd = GetTrueNodeDistance(neighborNode, endNode);
1018  neighborNode.totalDistance = neighborNode.distanceFromStart + neighborNode.trueDistanceFromEnd;
1019  neighborNode.parentNode = currentNode;
1020  openNodes.Insert(0, neighborNode);
1021  }
1022 
1023  nextNeighbor:{}
1024  }
1025  }
1026 
1027  // Create path
1028  createPath:
1029  List<PathNode> path = new List<PathNode>();
1030  while(currentNode != null) {
1031  path.Add(currentNode);
1032  currentNode = currentNode.parentNode;
1033  }
1034 
1035  //Debug.Log("Path");
1036  for(int i = 0; i < path.Count; i++) {
1037  //Debug.Log(path[i].position);
1038  }
1039 
1040  if(excludeNodesFromEnd > 0) {
1041  List<PathNode>pathFinal = new List<PathNode>();
1042  for(int i = 0; i < path.Count; i++) {
1043  if(i > excludeNodesFromEnd-1) {
1044  pathFinal.Add(path[i]);
1045  }
1046  }
1047  return pathFinal;
1048  } else {
1049  return path;
1050  }
1051  }
